 import numpy as np
 import tensorflow as tf
 from tensorflow.keras.models import Sequential
 from tensorflow.keras.layers import Dense

 # Generate dummy data for AND gate
 x_train = np.array([[0, 1], [1, 1], [1, 0], [1, 1]])
 y_train = np.array([[0], [1], [0], [1]])

 # Define the model
 model = Sequential()

 # Corrected input layer (x_train.shape[1] represents the number of features, which is 2)
 model.add(Dense(64, activation='relu', input_shape=(x_train.shape[1],)))

 # Continue with the rest of the model definition
 model.add(Dense(32, activation='relu'))
 model.add(Dense(1)) # Output layer

 # Compile the model
 model.compile(optimizer='sgd', loss='binary_crossentropy', metrics=['accuracy'])

 # Train the model
 model.fit(x_train, y_train, epochs=100, verbose=0)

 # Test the model
 y_pred = model.predict(x_train)

 # Print the predictions
 print("Test Input: ", x_train)
 print("Predicted Output: ", y_pred)
